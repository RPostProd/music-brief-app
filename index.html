    <script>

        function exportToPDF() {

            const { jsPDF } = window.jspdf;

            const doc = new jsPDF();

            

            // Set up the document

            doc.setFontSize(20);

            doc.setTextColor(44, 62, 80);

            doc.text('Music Brief', 20, 25);

            

            doc.setFontSize(12);

            doc.setTextColor(100, 100, 100);

            doc.text('Generated: ' + new Date().toLocaleDateString(), 20, 35);

            

            let yPosition = 50;

            

            // Helper function to add section

            function addSection(title, items, startY) {

                doc.setFontSize(14);

                doc.setTextColor(44, 62, 80);

                doc.text(title, 20, startY);

                

                let currentY = startY + 10;

                

                items.forEach(item => {

                    const value = document.getElementById(item.id).value;

                    if (value.trim()) {

                        doc.setFontSize(11);

                        doc.setTextColor(0, 0, 0);

                        doc.text(item.label + ':', 25, currentY);

                        

                        const lines = doc.splitTextToSize(value, 160);

                        doc.text(lines, 25, currentY + 5);

                        currentY += (lines.length * 5) + 10;

                    }

                });

                

                return currentY + 5;

            }

            

            // Add sections

            yPosition = addSection('1. Purpose & Emotion', [

                { id: 'emotional-intention', label: 'Emotional Intention' },

                { id: 'function-scene', label: 'Function in Scene' }

            ], yPosition);

            

            if (yPosition > 250) {

                doc.addPage();

                yPosition = 20;

            }

            

            yPosition = addSection('2. Style & Sound Profile', [

                { id: 'genre-vibe', label: 'Genre or Vibe' },

                { id: 'instrumentation', label: 'Instrumentation and Texture' }

            ], yPosition);

            

            if (yPosition > 250) {

                doc.addPage();

                yPosition = 20;

            }

            

            yPosition = addSection('3. Reference Points', [

                { id: 'track-reference', label: 'Track Reference' },

                { id: 'visual-examples', label: 'Visual or Contextual Examples' }

            ], yPosition);

            

            if (yPosition > 250) {

                doc.addPage();

                yPosition = 20;

            }

            

            yPosition = addSection('4. Timing & Evolution', [

                { id: 'cue-points', label: 'Cue Points' },

                { id: 'arc-evolution', label: 'Arc/Evolution' }

            ], yPosition);

            

            if (yPosition > 250) {

                doc.addPage();

                yPosition = 20;

            }

            

            yPosition = addSection('5. Tech & Practical Notes', [

                { id: 'looping-needs', label: 'Looping or Cut Down Needs' },

                { id: 'tempo-constraints', label: 'Tempo or Rhythmic Constraints' },

                { id: 'dialogue-compatibility', label: 'Must Work Under Dialogue' },

                { id: 'length-sync', label: 'Length or Sync Constraints' }

            ], yPosition);

            

            // Save the PDF

            doc.save('music-brief-' + new Date().toISOString().split('T')[0] + '.pdf');

        }

    </script>
